Цикл while

Полная форма

while (/*<условие>*/) { // Пока выполняется <условие> - выполняется <код>
    /*
        <код>
    */;
}

Краткая форма

while (/*<условие>*/) /*<код>*/;

// или
while (/*<условие>*/)
    /*<код>*/;

Пример

var i = 10;
while (i > 0) {
    if (i % 2 == 0)
        document.write(i);
    else document.write(i + " ");
    // & 109 87 65 43 21
    i--; // уменшаем i, чтобы не сделать бесконечный цикл 
}
